import { useEffect } from "react"; import { useLocation } from "react-router-dom"; export const useScroll = () => { const location = useLocation(); useEffect(() => { if (location.hash) { const id = location.hash.substring(1); const element = document.getElementById(id); if (element) { element.scrollIntoView({ behavior: "smooth" }); } } }, [location]); };
